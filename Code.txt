        private void A1_Click(object sender, EventArgs e)
        {
            Button b = (Button)sender;
            // CButton b = (Button)sender
            // Habe hiermit das Object, den Button, umgenannt in b
            // Dadurch kann ich schneller den Button als Object ansprechen weil er als 'b' angesprochen werden kann statt
            // Außerdem kann ich nun dieses Event allen Buttons zuweisen
            // da das Object des jeweiligen senders angesprochen wird und nicht der A1 Button jedes mal


            if (runde == true) // wenn true ist X an der Reihe
                 {
                     b.Text = "X";
                //b.Text = "X" setzt den Button text auf X
                     b.Enabled = false;
                //b.Enabled = false deaktiviert den Button sodass der Spieler nicht zweimal in ein Feld klicken kann
                     runde = !runde;
                //runde = !runde; setzt den vorherig definierten bool wert "runde" auf das gegenteil vom aktuellen wert. Damit beim nächsten drücken der Spieler O spielt
                button5.Visible = true;
                button4.Visible = false;
                //Spieler Button zum anzeigen wer dran ist
                 }

            else // wenn false ist O an der Reihe
                {
                     b.Text = "O";

                     b.Enabled = false;

                     runde = !runde;

                button5.Visible = false;
                button4.Visible = true;

                 }
            }
